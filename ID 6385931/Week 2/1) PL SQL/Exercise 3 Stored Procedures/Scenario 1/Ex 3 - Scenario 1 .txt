CREATE TABLE SavingsAccounts
(
  account_id int,
  balance float
);

CREATE OR REPLACE PROCEDURE ProcessMonthlyInterest is
BEGIN
  UPDATE SavingsAccounts
    SET balance=balance+(balance*0.01);
    for i in (SELECT acCount_id,balance from SavingsAccounts) LOOP
    DBMS_OUTPUT.PUT_LINE(i.account_id || ' Monthly Interest of 1% has been added ' || 'new balance: ' || TO_CHAR(i.balance));
    END LOOP;
    COMMIT;
END;
/

BEGIN
  ProcessMonthlyInterest;
END;
/
SELECT * FROM SavingsAccounts;
